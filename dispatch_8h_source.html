9// Given `i` calls `Cnv` callable with `i` as _template argument_ and passes the result into best
10// match of callable out of `fn` set.
11//
12// `i` has to fit in range 0..N-1, any other value is undefined behavior.
13template < std::size_t N, typename Cnv, typename... Fn >
14constexpr decltype( auto ) dispatch( index_type i, Cnv&& cnv, Fn&&... fn )
15{
16 using types = factory_result_types_t< N, Cnv >;
17
18 typename _check_unique_invocability< types >::template with_pure_value< Fn... > _{};
19
20 return _dispatch_index< 0, N >( i, [&]< index_type j >() -> decltype( auto ) {
21 auto&& item = cnv.template operator()< j >();
22
23 return _dispatch_fun( (decltype( item )&&) item, (Fn&&) fn... );
24 } );
25}
